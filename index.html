Пример кода

Задачу задали в одном из тестовых заданий, необходимо посчитать суммарную стоимость квартир в доме исходя из того, что каждые X этажей (delta) стоимость квартиры увеличивается на 1000, на одном этаже только одна квартира, параметром floorCount задается количество этажей в доме, параметром flatPrice задается первоначальная стоимость . 
                                 
 function housePrice(floorCount, delta, flatPrice) {
   let sum = 0;
   let floor = 1;
   let floordelta = 0;
   let sumFlatPrice = flatPrice;

   while (floor <= floorCount) {
     if (floordelta !== delta) {
       sum = sum + sumFlatPrice
       floor++
       floordelta++
     } else {
     sumFlatPrice = sumFlatPrice + 1000
     floordelta = 0
     }
   }
   return sum;
 };
